class Geeks_1
    
    def display (a, b)
	print (a, b)
    end
end
 
class Geeks_2 < Geeks_1

    def display (a, b)
         
        super
         
        super (a)
         
        super (a, b)
         
        super()
         
        print ("Hey! This is subclass method")
    end
end

obj = Geeks_2.new
 
obj.display ("Sudo_Placement", "GFG")
